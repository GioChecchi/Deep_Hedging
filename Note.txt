def generate_heston_paths_euler(S0, v0, r, kappa, theta, sigma_v, rho, T, steps, n_paths):
    """Fast Euler scheme for training — biased but 10x faster than exact CIR."""
    dt          = T / steps
    sqrt_dt     = np.sqrt(dt)
    sqrt_1_rho2 = np.sqrt(1.0 - rho**2)

    S  = np.full(n_paths, S0)
    v  = np.full(n_paths, v0)
    S2 = np.full(n_paths, _L(T, v, kappa, theta))
    running_integral = np.zeros(n_paths)

    S_out  = np.zeros((steps + 1, n_paths)); S_out[0]  = S
    v_out  = np.zeros((steps + 1, n_paths)); v_out[0]  = v
    S2_out = np.zeros((steps + 1, n_paths)); S2_out[0] = S2

    for step in range(steps):
        Z1 = np.random.randn(n_paths)
        Z2 = np.random.randn(n_paths)
        v_pos = np.maximum(v, 0.0)

        v_new = np.maximum(
            v + kappa * (theta - v_pos) * dt + sigma_v * np.sqrt(v_pos) * sqrt_dt * Z2, 0.0)
        I = 0.5 * (v_pos + v_new) * dt

        dB = rho * Z2 + sqrt_1_rho2 * Z1
        S  = S * np.exp(r * dt - 0.5 * I + np.sqrt(np.maximum(I, 0.0)) * dB)
        v  = v_new

        running_integral += I
        S2 = running_integral + _L(T - (step+1)*dt, v, kappa, theta)

        S_out[step+1]  = S
        v_out[step+1]  = v
        S2_out[step+1] = S2
